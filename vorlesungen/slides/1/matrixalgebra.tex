\begin{frame}[t]
\frametitle{Matrix-Algebra}
\vspace{-10pt}
\[
\begin{pmatrix}
a_{11}&\dots &a_{1n}\\
\vdots&\ddots&\vdots\\
a_{m1}&\dots &a_{mn}
\end{pmatrix}
+
\begin{pmatrix}
b_{11}&\dots &b_{1n}\\
\vdots&\ddots&\vdots\\
b_{m1}&\dots &b_{mn}
\end{pmatrix}
=
\begin{pmatrix}
a_{11}+b_{11}&\dots &a_{1n}+b_{1n}\\
\vdots&\ddots&\vdots\\
a_{m1}+b_{m1}&\dots &a_{mn}+b_{mn}
\end{pmatrix}
\]
\[
\lambda
\begin{pmatrix}
a_{11}&\dots &a_{1n}\\
\vdots&\ddots&\vdots\\
a_{m1}&\dots &a_{mn}
\end{pmatrix}
=
\begin{pmatrix}
\lambda a_{11}&\dots &\lambda a_{1n}\\
\vdots&\ddots&\vdots\\
\lambda a_{m1}&\dots &\lambda a_{mn}
\end{pmatrix}
\]
