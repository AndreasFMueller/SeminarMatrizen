\begin{block}{Zunahme}
Für alle $k<l$ gilt
\begin{align*}
\mathcal{J}^k(f) &\supsetneq \mathcal{J}^{k+1}(f)
\\
\mathcal{K}^k(f) &\subsetneq \mathcal{K}^{k+1}(f)
\end{align*}
Für $k\ge l$ gilt
\begin{align*}
\mathcal{J}^k(f) &= \mathcal{J}^{k+1}(f)
\\
\mathcal{K}^k(f) &= \mathcal{K}^{k+1}(f)
\end{align*}
Ausserdem ist $l\le n$
\end{block}
\end{column}
