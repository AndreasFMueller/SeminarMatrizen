\begin{block}{Eigenraum}
F체r $\lambda\in\Bbbk$ heisst
\begin{align*}
E_\lambda(f)
&=
\ker (f-\lambda)
\\
&=
\{v\in V\;|\; f(v) = \lambda v\}
\end{align*}
{\em Eigenraum} von $f$ zum Eigenwert $\lambda$.
\end{block}
$E_\lambda(f)\subset V$ ist ein Unterraum

\begin{block}{Eigenwert}
Falls $\dim E_\lambda(f)>0$ heisst $\lambda$ Eigenwert von $f$.
\end{block}

\end{column}
