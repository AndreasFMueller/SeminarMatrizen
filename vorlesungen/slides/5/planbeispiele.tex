\uncover<1->{%
\begin{block}{$f(z)$ analytisch\strut}
Beispiele:
\begin{align*}
e^z
&=
\sum_{k=0}^\infty \frac{z^k}{k!}
\\
\cos z
&=
\sum_{k=0}^\infty (-1)^k\frac{z^{2k}}{2k!}
\\
\sin z
&=
\sum_{k=0}^\infty (-1)^k\frac{z^{2k+1}}{(2k+1)!}
\end{align*}
\end{block}}
\end{column}
