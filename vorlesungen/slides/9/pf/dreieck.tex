\begin{columns}[t,onlytextwidth]
\begin{column}{0.32\textwidth}
\begin{block}{Satz}
\[
|u+v|\le |u|+|v|
\]
Gleichheit wenn lin.~abh.
\end{block}
\begin{block}{Satz}
\[
\biggl|\sum_i u_i\biggr|
\le
\sum_i |u_i|
\]
Gleichheit wenn $u_i = \lambda_i u$
\end{block}
\begin{block}{Satz}
\[
\biggl|\sum_i z_i\biggr|
\le
\sum_i |z_i|
\]
Gleichheit, wenn $z_i=|z_i|c$, $c\in\mathbb{C}$
\end{block}
\end{column}
