\begin{block}{Markov-Kette $\tilde{Y}$}
Übergangsmatrix
\[
\tilde{B}=
B+\varepsilon F
=
B+\varepsilon\begin{pmatrix*}[r]
0&1&-1\\
-1&0&1\\
1&-1&0
\end{pmatrix*}
\]
Gewinnmatrix:
\[
G=\begin{pmatrix*}[r]
0&-1&1\\
1&0&-1\\
-1&1&0
\end{pmatrix*}
\]
\end{block}
\begin{block}{Gewinnerwartung}
\begin{align*}
E(\tilde{Y})
&=
U^t(G\odot \tilde{B})p
\\
&=
E(Y) + \varepsilon U^t(G\odot F)p
=
\frac1{15}+2\varepsilon
\end{align*}
\end{block}
