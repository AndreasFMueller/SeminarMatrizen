\begin{block}{Markov-Kette $\tilde{Y}$}
Übergangsmatrix
\[
\tilde{B}=
B\uncover<2->{+\varepsilon F}
\uncover<3->{=
B+\varepsilon\begin{pmatrix*}[r]
0&1&-1\\
-1&0&1\\
1&-1&0
\end{pmatrix*}}
\]
\vspace{-12pt}

\uncover<4->{%
Gewinnmatrix:
\[
G=\begin{pmatrix*}[r]
0&-1&1\\
1&0&-1\\
-1&1&0
\end{pmatrix*}
\]}
\end{block}
\vspace{-12pt}
