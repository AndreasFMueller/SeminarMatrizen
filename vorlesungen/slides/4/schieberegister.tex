\bgroup
\def\ds{0.7}
\def\punkt#1#2{({(#1)*\ds},{(#2)*\ds})}
\def\rahmen{
	\draw ({-0.5*\ds},{-0.5*\ds}) rectangle ({7.5*\ds},{0.5*\ds});
	\foreach \x in {0.5,1.5,...,6.5}{
		\draw ({\x*\ds},{-0.5*\ds}) rectangle ({\x*\ds},{0.5*\ds});
	}
}
\def\polynom#1#2#3#4#5#6#7#8{
	\node at \punkt{0}{0} {$#1$};
	\node at \punkt{1}{0} {$#2$};
	\node at \punkt{2}{0} {$#3$};
	\node at \punkt{3}{0} {$#4$};
	\node at \punkt{4}{0} {$#5$};
	\node at \punkt{5}{0} {$#6$};
	\node at \punkt{6}{0} {$#7$};
	\node at \punkt{7}{0} {$#8$};
}
