\uncover<4->{%
\begin{block}{Allgemein}
\vspace{-12pt}
\begin{align*}
Y^2+XY &= X^3 + aX+b
\\
\uncover<5->{%
Y(Y+X) &= X^3 + aX + b}
\end{align*}
\uncover<6->{invariant unter}
\begin{align*}
\uncover<7->{X&\mapsto X,& Y&\mapsto -X-Y}
\\
\uncover<8->{&&\Rightarrow X+Y&\mapsto -Y}
\end{align*}
\uncover<9->{Spezialfall $\mathbb{F}_2$: $Y\leftrightarrow X+Y$}
\end{block}}
