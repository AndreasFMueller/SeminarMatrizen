\begin{block}{Endlicher Körper}
1 Byte = 8 bits: $\mathbb{F}_{2^8}$
mit Minimalpolynom:
\[
m(X) = X^8+X^4+X^3+X+1
\]
\end{block}
\vspace{-10pt}
\uncover<2->{%
\begin{block}{Inverse $a^{-1}$}
Mit dem euklidischen Algorithmus
\[
\begin{aligned}
sa+tm&=1
&&\Rightarrow&
\uncover<3->{
a^{-1} &= s}
\\
&
&&&
\uncover<4->{
\overline{a}
&=
\begin{cases}
a^{-1}&\; a\ne 0\\
0     &\; a =  0
\end{cases}}
\end{aligned}
\]
\end{block}}
\vspace{-10pt}
\uncover<5->{%
\begin{block}{Vektorraum}
$\mathbb{R}_{2^8}$
ist ein $8$-dimensionaler $\mathbb{F}_2$-Vektorraum
\end{block}}
\end{column}
