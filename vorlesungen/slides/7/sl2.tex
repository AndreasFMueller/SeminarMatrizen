\begin{block}{Determinante}
\[
A=\begin{pmatrix}
a&b\\
c&d
\end{pmatrix}
\;\Rightarrow\;
\det A = ad-bc
\]
\end{block}

\begin{block}{Dimension}

3\; \text{Dimensionen}
\]
\end{block}
