\begin{align*}
\uncover<2->{
D_{x,\alpha}

\begin{pmatrix}
1&0&0\\0&\cos\alpha&-\sin\alpha\\0&\sin\alpha&\cos\alpha
\end{pmatrix}
}
\\
\uncover<3->{
D_{y,\beta}
&=
\begin{pmatrix}
\cos\beta&0&\sin\beta\\0&1&0\\-\sin\beta&0&\cos\beta
\end{pmatrix}
}
\\
\uncover<4->{
D_{z,\gamma}
&=
\begin{pmatrix}
\cos\gamma&-\sin\gamma&0\\\sin\gamma&\cos\gamma&0\\0&0&1
\end{pmatrix}
}
\intertext{\uncover<5->{beliebige Drehung:}}
\uncover<5->{
D
&=
D_{x,\alpha}
D_{y,\beta}
D_{z,\gamma}
}
\end{align*}
