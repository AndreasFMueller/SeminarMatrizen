\newboolean{pfeilspitzen}

\def\knoten#1#2{
        \fill[color=white] #1 circle[radius=0.3];
        \draw[line width=1pt] #1 circle[radius=0.3];
        \node at #1 {$#2$};
}

\def\kante#1#2#3{
        \ifthenelse{\boolean{pfeilspitzen}}{
                \draw[->,line width=1pt,shorten >= 0.3cm,shorten <= 0.3cm]
                        #1 -- #2;
        }{
                \draw[line width=1pt,shorten >= 0.3cm,shorten <= 0.3cm]
                        #1 -- #2;
        }
