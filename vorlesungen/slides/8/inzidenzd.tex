\begin{block}{Definition}
\vspace{-20pt}
\begin{align*}
B(G)_{ij}&=-1&&\Leftrightarrow&&\text{Kante $j$ von $i$}\\
B(G)_{kj}&=+1&&\Leftrightarrow&&\text{Kante $j$ nach $k$}\\
A(G)_{ij}&=\phantom{-}1&&\Leftrightarrow&&\text{Kante von $i$ nach $j$}
\end{align*}
\end{block}
