\def\kante#1#2{
	%\fill[color=white,opacity=0.5] #1 circle[radius=0.2];
	\fill[color=white,opacity=0.5] 
		($#1+(-\s,-\s)$) --
		($#1+(+\s,-\s)$) --
		($#1+(+\s,+\s)$) --
		($#1+(-\s,+\s)$) -- cycle;
	\node at #1 {$#2$};
}

\def\flaeche#1#2{
	\fill[color=white,opacity=0.5]
		($#1+({-\r*cos(30)},{-\r*sin(30)})$) --
		($#1+({\r*cos(30)},{-\r*sin(30)})$) --
		($#1+(0,{\r})$) -- cycle;
	\node at #1 {$#2$};
}
