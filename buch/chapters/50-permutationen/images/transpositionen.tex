\def\kreuz#1#2#3{
	\draw[->] ({(#1)-1},#2) to[out=-90,in=90] ({#1},{#2-1});
	\draw[->] ({#1},#2) to[out=-90,in=90] ({#1-1},{#2-1});
	\node at ({(#1)-0.5+0.2},{#2-0.5}) [right] {$#3$};
}
